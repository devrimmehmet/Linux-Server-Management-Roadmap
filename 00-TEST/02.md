Script dosyasını oluştur:

```bash
nano setup.sh
```

```bash
#!/bin/bash
set -e

echo "🚀 MSSQL Server 2022 Kurulumu Başlıyor (Ubuntu 22.04.5 LTS - Jammy Jellyfish)..."

# 1. Güncelleme

sudo apt update -y
sudo apt upgrade -y

# 2. Microsoft GPG key ekle

curl -sSL https://packages.microsoft.com/keys/microsoft.asc | sudo gpg --dearmor -o /usr/share/keyrings/microsoft.gpg

# 3. MSSQL Server repo ekle (Ubuntu 22.04 - jammy)

echo "deb [arch=amd64 signed-by=/usr/share/keyrings/microsoft.gpg] https://packages.microsoft.com/ubuntu/22.04/mssql-server-2022 jammy main" | sudo tee /etc/apt/sources.list.d/mssql-server.list

# 4. MSSQL Tools repo ekle

echo "deb [arch=amd64 signed-by=/usr/share/keyrings/microsoft.gpg] https://packages.microsoft.com/ubuntu/22.04/prod jammy main" | sudo tee /etc/apt/sources.list.d/msprod.list

# 5. Paket listesi güncelle

sudo apt update -y

# 6. MSSQL Server kur

sudo apt install -y mssql-server

# 7. MSSQL Tools (sqlcmd, bcp) kur

sudo apt install -y mssql-tools unixodbc-dev

# sqlcmd PATH ekle

if ! grep -q "/opt/mssql-tools/bin" ~/.bashrc; then
echo 'export PATH="$PATH:/opt/mssql-tools/bin"' >> ~/.bashrc
fi
source ~/.bashrc

# 8. MSSQL Server setup (interaktif)

echo "⚠️ Şimdi MSSQL Server kurulumu yapılandırılacak."
echo "👉 Edition seçin (Developer için 2)"
echo "👉 SA şifresini girin (örn: dMp@2025!)"
sudo /opt/mssql/bin/mssql-conf setup

# 9. MSSQL servisini etkinleştir

sudo systemctl enable mssql-server
sudo systemctl restart mssql-server

echo "✅ MSSQL Server kurulumu tamamlandı!"
echo "🔑 sqlcmd ile bağlanmak için: sqlcmd -S localhost -U SA -P '<Parolan>'"
echo "🌐 Dış bağlantı için port açmayı unutmayın: sudo ufw allow 1433/tcp"
```
